#Choosing Shroudwalker Enclave To Generate
choose_shroudwalker_enclave_country = {
	if = {
		limit = {
			NOT = { any_country = { has_origin = origin_imperial_vassal } }
		}
		if = {
			limit = {
				any_system = { has_star_flag = mraxist }
			}
			set_planet_flag = enclave_dialectics
		}
		else = {
			set_planet_flag = enclave_shroudwalker_default
		}
	}
	else = {
		event_target:shroudwalker_enclave_system = {
			choose_imperial_overlord_country = yes
		}
		if = {
			limit = {
				any_system = { has_star_flag = mraxist }
			}
			set_planet_flag = enclave_dialectics
		}
		else_if = {
			limit = {
				any_system = { has_star_flag = is_chosen_overlord_soviet }
			}
			if = {
				limit = {
					AND = {
						galaxy_size = tiny
						NOT = { any_system = { has_star_flag = international } }
					}
				}
				set_planet_flag = enclave_shroudwalker_default
			}
			else = {
				set_planet_flag = enclave_dialectics
			}
		}
		else = {
			set_planet_flag = enclave_shroudwalker_default
		}
	}
}

#Choosing FOTD Korea To Generate
choose_fear_of_the_dark_country_korea = {
	set_variable = {
		which = counted_corporations
		value = 0
	}
	every_playable_country = {
		limit = {
			has_authority = auth_corporate
		}
		prev = {
			change_variable = {
				which = counted_corporations
				value = 1
			}
		}
	}
	if = {
		limit = {
			galaxy_size = small
		}
		if = {
			limit = {
				check_variable_arithmetic = {
					which = counted_corporations
					value < 3
				}
			}
			set_planet_flag = north
		}
		else = {
			set_planet_flag = south
		}
	}
	else_if = {
		limit = {
			galaxy_size = medium
		}
		if = {
			limit = {
				check_variable_arithmetic = {
					which = counted_corporations
					value < 4
				}
			}
			set_planet_flag = north
		}
		else = {
			set_planet_flag = south
		}
	}
	else_if = {
		limit = {
			galaxy_size = large
		}
		if = {
			limit = {
				check_variable_arithmetic = {
					which = counted_corporations
					value < 5
				}
			}
			set_planet_flag = north
		}
		else = {
			set_planet_flag = south
		}
	}
	else_if = {
		limit = {
			galaxy_size = huge
		}
		if = {
			limit = {
				check_variable_arithmetic = {
					which = counted_corporations
					value < 6
				}
			}
			set_planet_flag = north
		}
		else = {
			set_planet_flag = south
		}
	}
	else = {
		if = {
			limit = {
				check_variable_arithmetic = {
					which = counted_corporations
					value < 2
				}
			}
			set_planet_flag = north
		}
		else = {
			set_planet_flag = south
		}
	}
	clear_variable = counted_corporations
}

	if = {
		limit = {
			any_neighbor_system = {
				has_owner = no
				has_special_star_flag_trigger = no
			}
		}
		random_neighbor_system = {
			limit = {
				has_owner = no
				has_special_star_flag_trigger = no
			}
			set_star_flag = overlord_empire_flag
			set_star_flag = used_as_a_way_out
			spawn_system = {
				min_distance >= 20
				max_distance <= 35
				initializer = overlord_system_9_init
				hyperlane = yes

				effect = {
					set_star_flag = overlord_empire_flag
					set_star_flag = used_as_a_way_out
				}
			}
		}
	}
	else_if = {
		limit = {
			any_neighbor_system_euclidean = {
				has_owner = no
				has_special_star_flag_trigger = no
			}
		}
		random_neighbor_system_euclidean = {
			limit = {
				has_owner = no
				has_special_star_flag_trigger = no
			}
			add_hyperlane = {
				from = this
				to = prev
			}
			set_star_flag = overlord_empire_flag
			set_star_flag = used_as_a_way_out
			spawn_system = {
				min_distance >= 20
				max_distance <= 35
				initializer = overlord_system_9_init
				hyperlane = yes

				effect = {
					set_star_flag = overlord_empire_flag
					set_star_flag = used_as_a_way_out
				}
			}
		}
	}
	else = {
		spawn_system = {
			min_distance >= 20
			max_distance <= 35
			initializer = overlord_system_9_init
			hyperlane = yes

			effect = {
				set_star_flag = overlord_empire_flag

		}
		else = {
			create_imperial_splinter_random = yes
		}
	}
	else = {
		random_list = {

					any_country = { has_country_flag = strelkov }
				}

			}

			}

			}

					factor = 0
					any_country = { has_country_flag = kasparov }
				}
				create_imperial_splinter_kasparov = yes
			}
			144 = { # kurginyan
				modifier = {
					factor = 0
					any_country = { has_country_flag = kurginyan }
				}
				create_imperial_splinter_kurginyan = yes
			}

				create_imperial_splinter_rdc = yes
			}
			144 = { # but
				modifier = {
					factor = 0
					any_country = { has_country_flag = but }
				}
				create_imperial_splinter_but = yes
			}
			144 = { # hodorkovsky
				modifier = {
					factor = 0
					any_country = { has_country_flag = hodorkovsky }
				}
				create_imperial_splinter_hodorkovsky = yes
			}
			144 = { # abramovich
				modifier = {
					factor = 0
					any_country = { has_country_flag = abramovich }
				}
				create_imperial_splinter_abramovich = yes
			}
			1 = {
				create_imperial_splinter_random = yes
			}

	set_owner = last_created_country

		every_system_planet = {
			limit = {
				AND = {
					is_colony = yes
					is_controlled_by = root
				}

			set_owner = last_created_country
		}

		limit = {
			has_communications = root

	}

		limit = {
			NOT = { has_communications = root }
			has_active_first_contact_with = root
		}

			who = last_created_country
			location = last_created_country.capital_star
		}
	}

	every_playable_country = {
		limit = {
			has_country_flag = imperial_splinter
			NOT = { is_same_empire = last_created_country }
		}
		establish_communications_no_message = last_created_country
	}

check_country_flags_mad_vacuum_cleaner = {
	if = {
		limit = { NOT = { has_country_flag = mad_vacuum_cleaner_hired } }
		set_country_flag = mad_vacuum_cleaner_hired
	}

	if = {
		limit = { any_owned_leader = { has_trait = leader_trait_tomcat_insight } }
		set_country_flag = has_mad_vacuum_cleaner
	}
	else_if = {
		limit = { has_country_flag = has_mad_vacuum_cleaner }
		remove_country_flag = has_mad_vacuum_cleaner
	}

	if = {
		limit = { any_owned_species = { is_feline_species = yes } }
		set_country_flag = has_feline_species
		if = {
			limit = { owner_main_species = { is_feline_species = yes } }
			set_country_flag = feline_country
			if = {
				limit = {
					is_imperial_authority = yes
					exists = ruler
					ruler = { species = { is_feline_species = yes } }
				}
				set_country_flag = feline_monarchy
			}
			else_if = {
				limit = { has_country_flag = feline_monarchy }
				remove_country_flag = feline_monarchy
			}
		}
		else_if = {
			limit = { has_country_flag = feline_country }
			remove_country_flag = feline_country
		}
	}
	else_if = {
		limit = { has_country_flag = has_feline_species }
		remove_country_flag = has_feline_species
	}

	if = {
		limit = {
			any_planet_within_border = {
				any_owned_pop_group = {
					is_sapient = no
					is_feline_species = yes
					NOT = { has_trait = trait_nascent_stage }
				}
			}
		}
		set_country_flag = feline_pre_sapients
	}

	if = {
		limit = {
			has_country_flag = has_mad_vacuum_cleaner
			has_country_flag = has_feline_species
			NOT = { has_country_flag = xenoist_enabled }
		}
		set_country_flag = xenoist_enabled
	}
}

	}

	#}
	else_if = {
		limit = {

				has_trait = leader_trait_destructive
				has_trait = leader_trait_destructive_2
			}
		}
		add_trait = leader_trait_destructive
	}
	else_if = {
		limit = { NOT = { has_trait = leader_trait_destructive_2 } }
		add_trait = leader_trait_destructive_2
	}
	else_if = {

embrace_catist_faction_no_name_change = {
	if = {
		limit = {
			ruler = { species = { is_feline_species = yes } }
		}
		ruler = {
			species = {
				save_event_target_as = new_dominant_species
			}
		}
	}
	else_if = {
		limit = {
			any_owned_species = {
				is_feline_species = yes
				OR = {
					is_sapient = yes
					has_trait = trait_nascent_stage
				}
				is_robotic = no
			}
		}
		random_owned_species = {
			limit = {
				is_feline_species = yes
				OR = {
					is_sapient = yes
					has_trait = trait_nascent_stage
				}
				is_robotic = no
			}
			save_event_target_as = new_dominant_species
		}
	}
	else_if = {
		limit = {
			any_owned_species = {
				is_feline_species = yes
				OR = {
					is_sapient = yes
					has_trait = trait_nascent_stage
				}
			}
		}
		random_owned_species = {
			limit = {
				is_feline_species = yes
				OR = {
					is_sapient = yes
					has_trait = trait_nascent_stage
				}
			}
			save_event_target_as = new_dominant_species
		}
	}
	else_if = {
		limit = {
			any_owned_species = {
				is_feline_species = yes
			}
		}
		random_owned_species = {
			limit = {
				is_feline_species = yes
			}
			save_event_target_as = new_dominant_species
		}
	}
	else = { # should never be triggered
		owner_main_species = { save_event_target_as = new_dominant_species }
	}

	change_dominant_species = { species = event_target:new_dominant_species change_all = yes }

	country_add_ethic = ethic_authoritarian
	country_add_ethic = ethic_xenophobe
	country_add_ethic = ethic_militarist
	change_government = {
		authority = auth_imperial
		remove_invalid_civics = yes
		civics = {
			civic = civic_philosopher_king
			civic = civic_pleasure_seekers
			#civic = civic_crafters
		}
	}
	
	if = {
		limit = {
			OR = {
				NOT = { exists = ruler }
				AND = {
					exists = ruler
					ruler = { species = { NOT = { is_feline_species = yes } } }
				}
			}
		}
		create_leader = {
			# All Hail Gorongoron!
			species = event_target:new_dominant_species
			class = scientist
			name = "Горонгорон"
			gender = male
			skill = 2
			set_age = 32
			traits = {
				trait = trait_ruler_charismatic_2
			}
		}
		set_leader = last_created_leader
	}
	set_male_ruler_title = "Его Котейшество"
	set_female_ruler_title = "Её Котейшество"
	set_male_heir_title = "Котей"
	set_female_heir_title = "Котесса"

	check_country_flags_mad_vacuum_cleaner = yes
}
